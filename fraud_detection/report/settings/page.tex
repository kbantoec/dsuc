% Title format ------------------------------------------------------

%\titleformat{command}[shape]{format}{label}{sep}{before-code}{after-code}
% command: \part, \chapter, \section, \subsection, \subsubsection, \paragraph or \subparagraph.
% shape: hang, block, display, runin, leftmargin, rightmargin, drop, wrap or frame.
% format: is the format to be applied to the title, label, and text; for example \normalfont\Large\bfseries.

% Further information:
% https://www.overleaf.com/learn/latex/Sections_and_chapters
% link: https://www.youtube.com/watch?v=VRp4B1JzHTM

\titleformat
{\part} % command
[display] % shape
{\large\center} % format
{\sffamily\textbf{\partname\ \thepart}} % label
{20pt} % separation between chapter and label
{\Huge\itshape} % title label options

\titleformat
{\chapter} % command
[display] % shape
{\large\itshape\center} % format
{\chaptertitlename~\thechapter} % label
{10pt} % separation between chapter and label
{\huge} % title label options

% \titlespacing{command}{left}{before-sep}{after-sep}

\titleformat
{\section}
[hang]
{\Large\sffamily\bfseries}
{\thesection}
{0.5em}
{}

\titleformat
{\subsection}
[hang]
{\large\sffamily\bfseries}
{\thesubsection}
{0.5em}
{}
